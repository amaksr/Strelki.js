  9     /**
 10      * Create new IndexArray object
 11      * @class
 12      * IndexedArray is a hybrid array with multiple hash indexes.
 13      * It can have additional "indexes": hash objects that point to elements and that can be quickly accessed by index value.
 14      * Elements also can be accessed by position.
 15      * @name IndexedArray
 16      * @constructor
 17      */
 18     var IndexedArray = function() {
 19         this.data = {};
 20         this.data_ids = null;
 21         this.indexData = {};
 22         this.ordered = false;
 23     }
 24 
 25     /**
 26      * Set flag that indicates whether array should be ordered by "id" or not
 27      * @param {boolean} ordered
 28      */
 29     IndexedArray.prototype.setOrdered = function(ordered) {
 30         this.ordered = ordered;
 31         if(this.ordered && this.data_ids)
 32             this.data_ids.sort();
 33     };
 34 
 35     IndexedArray.prototype._getDataIds = function() {
 36         if(!this.data_ids) {
 37             this.data_ids = Object.keys(this.data);
 38             if(this.ordered)
 39                 this.data_ids.sort();
 40         }
 41         return this.data_ids;
 42     };
 43 
 44     /**
 45      * Creates hash index on array using fieldName property of the elements
 46      * @param {String} fieldName
 47      */
 48     IndexedArray.prototype.createIndex = function(fieldName) {
 49         if(fieldName === 'id')
 50             return;
 51 
 52         var idxData = {};
 53         var dataKeys = this._getDataIds();
 54         for(var i=0; i<dataKeys.length; i++) {
 55             var id = dataKeys[i];
 56             var e = this.data[id];
 57             var fv = e[fieldName];
 58             if(!idxData[fv])
 59                 idxData[fv] = {};
 60 
 61             idxData[fv][id] = 1;
 62         }
 63         this.indexData[fieldName] = idxData;
 64     };

 67      * Returns hash array of objects
 68      * @returns {{}|*}
 69      */
 70     IndexedArray.prototype.getData = function() {
 71         return this.data;

 75      * Puts all the elements from array

 78     IndexedArray.prototype.loadArray = function(array) {
 79         if(!array || !array.length)
 80             return;
 81         for(var i=0; i<array.length; i++)

138     /**
139      * Stores element with unique "id" to array. If element with given "id" was already there, it will be overriden with the new element
140      * @param element
141      */
142     IndexedArray.prototype.put = function(element) {
143         this.del(element.id);
144         this.data_ids = null;
145         this.data[element.id] = element;
146 
147         var indexes = Object.keys(this.indexData);
148         for(var i = 0; i < indexes.length; i++) {
149             var fld = element[indexes[i]];
150             var idxData = this.indexData[indexes[i]];
151             if(!idxData[fld])
152                 idxData[fld] = {};
153 
154             idxData[fld][element.id] = 1;
155         }
156     };
157 
158     /**
159      * Delete element by "id"
160      * @param id
161      */
162     IndexedArray.prototype.del = function(id) {
163         var e = this.data[id];
164         if(!e)
165             return;
166         var indexes = Object.keys(this.indexData);
167         for(var i = 0; i < indexes.length; i++) {
168             var idxData = this.indexData[indexes[i]];
169             var idxElements  = idxData[e[indexes[i]]];
170             if(idxElements) {
171                 delete idxElements[id];
172                 if(Object.keys(idxElements).length === 0)
173                     delete idxData[e[indexes[i]]];
174             }
175         }
176         delete this.data[id];
177         this.data_ids = null;
178     };

201     /**
202      * Find elements by index value, and return their ids
203      * @param {String} filedName - index name
204      * @param {String} value - index value
205      * @returns {Array} - array of ids, empty array if not found
206      */
207     IndexedArray.prototype.findIdsByIndex = function(filedName,value) {
208         if(filedName === 'id') {
209             var val = this.data[value];
210             if(val)
211                 return [val.id];
212             else
213                 return [];
214         }
215         else {
216             var idxData = this.indexData[filedName];
217             if(!idxData)
218                 throw "Index " + filedName + " does not exist";
219             if(!idxData[value])
220                 return [];
221 
222             return Object.keys(idxData[value]);
223         }
224     };

232     /**
233      * For element el perform lookups according to joinInfoArray, and return all related records
234      * @param el - element
235      * @param {Array} joinInfoArray
236      */
237     IndexedArray.doLookups = function(el, joinInfoArray) {
238         var res=[];
239         for(var i=0; i<joinInfoArray.length; i++) {
240             var ji = joinInfoArray[i];
241             var joins = IndexedArray.doLookup(el, ji);
242             res = IndexedArray._merge(res,joins);
243         }
244         return res;
245     };
246 
247     IndexedArray.doLookup = function(el, joinInfo) {
248         var res = [];
249         var from_col = joinInfo.from_col;
250         var to_table = joinInfo.to_table;
251         var to_col = joinInfo.to_col;
252         var to_ids = el?to_table.findIdsByIndex(to_col,el[from_col]):[];
253 
254         if(to_ids.length == 0 && joinInfo.type === 'outer')
255             to_ids.push(null);
256 
257         for(var i=0; i < to_ids.length; i++) {
258             var to_id = to_ids[i];
259             var to_el = to_id?to_table.getData()[to_id]:null;
260             if(joinInfo['join']) {
261                 var joins = IndexedArray.doLookups(to_el,joinInfo['join']);
262                 for(var j=0; j<joins.length; j++)
263                     res.push([to_el].concat(joins[j]));
264             }
265             else
266                 res.push([to_el]);
267         }
268 
269         return res;
270     };

272     /**
273      * Join other IndexedArrays according to joinInfoArray
274      * <pre>
275      * joinInfoArray = [
276      *  	{
277 	 *  		from_col:   mandatory field name in 'el' element
278 	 *  		to_table:   mandatory reference to IndexedArray
279 	 *  		to_col:     mandatory id or other indexed field in referenced IndexedArray
280 	 *  	        type:       'outer' for outer join, null for inner join
281 	 *  	        join:       optional nested joinInfoArray structure
282 	 *  	},
283      *
284      *  ]
285      *  </pre>
286      * @param joinInfoArray
287      * @returns {Array} - array of records, where each record is an array of joined elements
288      */
289     IndexedArray.prototype.query = function(joinInfoArray) {
290         var res = [];
291         var mainKeys = this._getDataIds();
292         for(var i=0; i < mainKeys.length; i++) {
293             var el = this.data[mainKeys[i]];
294             var joins = IndexedArray.doLookups(el,joinInfoArray);
295 
296             for(var j=0; j<joins.length; j++)
297                 res.push([el].concat(joins[j]));
298         }
299         return res;
300     };

302     /**
303      * Search array by index, and then by filterCallback function
304      * @param fieldName - name of the index, null to select all elements
305      * @param value - value of the index
306      * @param filterCallback - filter function that eccepts element as a parameter, and returns true or false
307      * @returns {IndexedArray} - new IndexedArray with selected records
308      */
309     IndexedArray.prototype.where = function(fieldName, value, filterCallback) {
310         var res = new IndexedArray();
311 
312         var ids;
313 
314         if(fieldName)
315             ids = this.findIdsByIndex(fieldName, value);
316         else
317             ids = this._getDataIds();
318 
319         for(var i = 0; i < ids.length; i++) {
320             var el = this.get(ids[i]);
321             if(!filterCallback || filterCallback && filterCallback(el))
322                 res.put(el);
323         }
324         return res;
325     };

327     IndexedArray._merge = function(arr1, arr2) {
328         if(arr1.length === 0)
329             return arr2;
330         var res=[];
331         for(var i=0; i < arr1.length; i++) {
332             for(var j=0; j < arr2.length; j++) {
333                 res.push(arr1[i].concat(arr2[j]));
334             }
335         }
336         return res;
337     };
